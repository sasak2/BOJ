int n,m;
int arr[10];

void print(void) {
	for(int i=0; i<m; i++) {
		printf("%d ",arr[i+1]);
	}
	printf("\n");
}
void f(int depth) {
	if(depth==m+1) 
		print();
	if(arr[depth-1]==n)
		return;

	for(int i=arr[depth-1]+1; i<=n; i++) {
		arr[depth]=i;
		f(depth+1);
	}

	
	
}
