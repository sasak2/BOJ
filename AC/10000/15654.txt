int n,m;
int arr[10];
int num[10];

void print(void) {
	for(int i=0; i<m; i++) {
		printf("%d ",num[arr[i+1]-1]);
	}
	printf("\n");
}

void f(int depth) {
	if(depth==m+1)  {
		print();
		return;
		}
	

	for(int i=1; i<=n; i++) {
		int d=0;
		for(int k=1; k<depth;k++)
		if(arr[k]==i) {
			d=1;
			break;
		}
		
		if(d==1)
			continue;
	
		arr[depth]=i;
		f(depth+1);

		
	}
 
	
	
}

int main() {
	freopen("input.txt","rt",stdin);

	scanf("%d %d",&n,&m);

	for(int i=0; i<n; i++)
		scanf("%d",&num[i]);

	sort(num,num+n);
	
	arr[0]=0;
	f(1);

	

	return 0;
}
